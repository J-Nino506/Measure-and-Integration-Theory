    \begin{enumerate}[label = (\alph{*})]
    
        \item Let \( f \) be a bounded function \( [a,b] \), and let \( h \) be the upper envelope of \( f \) (cf. Problem 2.51). Then \( \displaystyle R \lowint{a}{b} f = \int_{a}^{b} h \).
        \item Use part (a) to prove Proposition 7 which is stated as follows
            \begin{prop}[\textbf{4.7}]
                A bounded function \( f \) on \( [a,b] \) is Riemann integrable if and only if the set of points at which \( f \) is discontinuous has measure zero. 
                
            \end{prop}
    \end{enumerate}


\end{problem}

\begin{problem}[4.3]

    Let \( f \) be a nonnegative measurable function. Show that \( \displaystyle f = 0 \) implies \( f = 0 \) almost everywhere.

            \int \varliminf_{n \to \infty} f_n \leq \varliminf_{n \to \infty} \int f_n.  
        \]
    
\end{problem}
